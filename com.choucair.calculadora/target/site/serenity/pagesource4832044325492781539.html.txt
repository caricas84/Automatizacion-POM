//	ÍNDICE DE MASA CORPORAL
function cal_bmi(lbs, ins){
   h2 = ins * ins;
   bmi = lbs/h2 * 703
   f_bmi = Math.floor(bmi);
   diff  = bmi - f_bmi;
   diff = diff * 10;
   diff = Math.round(diff);
   if (diff == 10){
      // Need to bump up the whole thing instead
      f_bmi += 1;
      diff = 0;
   }
   bmi = f_bmi + "." + diff;
   return bmi;
}

<h2>Cómo funciona la calculadora de calorías diarias</h2>
<p>El funcionamiento de esta <a href="https://es.calcuworld.com/">calculadora online</a> es muy sencillo.
